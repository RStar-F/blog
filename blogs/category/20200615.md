# rollup打包ts项目

安装相应的依赖

```javascript
// 安装rollup依赖
npm install -D rollup rollup-plugin-babel rollup-plugin-terser rollup-plugin-typescript2 typescript  
// 安装babel依赖
npm install -D @babel/core @babel/preset-env @babel/preset-typescript

```



配置tsconfig.json

```json
{
  "compilerOptions": {
    "target": "esnext",
    "module": "esnext",
    "strict": true,
    "jsx": "preserve",
    "importHelpers": true,
    "declaration": true, // 自动生成 xx.d.ts
    "declarationDir": "./lib", // xx.d.ts文件存储的路径
    "moduleResolution": "node",
    "skipLibCheck": true,
    "esModuleInterop": true,
    "allowSyntheticDefaultImports": true,
    "sourceMap": true,
    "baseUrl": ".",
    "lib": [
      "esnext",
      "dom",
      "dom.iterable",
      "scripthost"
    ]
  },
  "include": [
    "*.ts"
  ],
  "exclude": [
    "node_modules"
  ]
}
```



配置babel

```javascript
{
  "presets": [
    "@babel/preset-env",
    "@babel/preset-typescript"
  ]
}
```



配置rollup.config.js

```javascript
import { terser } from 'rollup-plugin-terser';
import typescript from 'rollup-plugin-typescript2';
const path = require('path');
const babel = require('rollup-plugin-babel');
const pkg = require('./package.json');

const extensions = ['.js', '.ts'];

const resolve = function(...args) {
  return path.resolve(__dirname, ...args);
};

module.exports = {
  input: resolve('./index.ts'),
  output: {
    file: resolve('./', pkg.main), // 为了项目的统一性，这里读取 package.json 中的配置项
    format: 'iife',
  },
  plugins: [
    typescript({
      tsconfig: './tsconfig.json'  // 这里是对ts处理的一些配置，使用tsconfig的配置，这样可以自动生成.d.ts文件
    }),
    babel({
      exclude: 'node_modules/**',
      extensions,
    }),
    terser()
  ],
};
```
